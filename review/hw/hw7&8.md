3. 字符串、指针、数组

   ```c
   int isPalindrome(char *string) {
       int len = strlen(string);
       for (int i = 0; i < len / 2; i++) {
           if (string[i] != string[len - i - 1])
               return 0;
       }
       return 1;
   }
   ```
